006package jdraw.std;
007
008import jdraw.framework.DrawCommand;
009import jdraw.framework.DrawCommandHandler;
010
011/**
012 * Provides an empty command handler. This class provides an empty dummy implementation of the draw command
013 * handler. It enables the application to start up correctly and to behave meaningful, but with the limitation
014 * that it does not provide any undo/redo behavior. 
015 * @author Christoph. Denzler
016 *
017 */
018public class EmptyDrawCommandHandler implements DrawCommandHandler {
019
020        @Override
021        public void addCommand(DrawCommand cmd) { /* do nothing. */ }
022        
023        @Override
024        public void undo() { /* do nothing. */ }
025
026        @Override
027        public void redo() { /* do nothing. */ }
028
029        @Override
030        public boolean undoPossible() { return false; }
031
032        @Override
033        public boolean redoPossible() { return false; }
034
035        @Override
036        public void beginScript() { /* do nothing. */ }
037
038        @Override
039        public void endScript() { /* do nothing. */ }
040
041        @Override
042        public void clearHistory() { /* do nothing. */ }
043}
