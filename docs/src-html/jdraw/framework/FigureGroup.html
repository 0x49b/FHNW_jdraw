008/**
009 * This interface provides access to figure parts for those figures 
010 * which consist of several parts, as e.g. for group figures.
011 *
012 * @author  Dominik Gruntz &amp; Christoph Denzler
013 * @version 2.5
014 */
015public interface FigureGroup {
016        
017        /**
018         * Returns the parts of a figure which consists of several parts.
019         * 
020         * @return parts of a figure
021         */
022        Iterable<Figure> getFigureParts();
023}
