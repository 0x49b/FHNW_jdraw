006package jdraw.framework;
007
008import java.awt.Cursor;
009import java.awt.Graphics;
010import java.awt.Point;
011import java.awt.event.MouseEvent;
012
013/**
014 * Handles are used to change a figure by direct manipulation. Handles know
015 * their owning figure and they provide methods to ask the handle's bounds and
016 * to track changes.
017 * 
018 * @see Figure
019 * 
020 * @author Dominik Gruntz &amp; Christoph Denzler
021 * @version 2.5
022 */
023public interface FigureHandle {
024
025        /**
026         * Gets the handle's owner.
027         * 
028         * @return the figure which owns the handle
029         */
030        Figure getOwner();
031
032        /**
033         * Returns the location of this handle. The result depends on the location
034         * of the owning figure.
035         * 
036         * @return location of this handle
037         */
038        Point getLocation();
039
040        /**
041         * Draws this handle.
042         * 
043         * @param g the graphics context to use for painting.
044         */
045        void draw(Graphics g);
046
047        /**
048         * Returns a curser which should be displayed when the mouse is over the
049         * handle. Signals the type of operation which can be performed using this
050         * handle.
051         * <P>
052         * A default implementation may return Cursor.getDefaultCursor().
053         * 
054         * @return handle's Cursor
055         */
056        Cursor getCursor();
057
058        /**
059         * Tests if a point is contained in the handle.
060         * 
061         * @param x x-coordinate of mouse position
062         * @param y y-coordinate of mouse position
063         * @return <code>true</code>, if coordinates are contained in the figure,
064         *         <code>false</code> otherwise
065         */
066        boolean contains(int x, int y);
067
068        /**
069         * Tracks the start of an interaction. Usually, the position where an
070         * interaction starts is stored.
071         * 
072         * @param x the x position where the interaction started
073         * @param y the y position where the interaction started
074         * @param e the mouse event which initiated the start interaction
075         * @param v the view in which the interaction is performed
076         */
077        void startInteraction(int x, int y, MouseEvent e, DrawView v);
078
079        /**
080         * Tracks a step of a started interaction.
081         * 
082         * @param x the current x position
083         * @param y the current y position
084         * @param e the mouse event which initiated the drag interaction
085         * @param v the view in which the interaction is performed
086         */
087        void dragInteraction(int x, int y, MouseEvent e, DrawView v);
088
089        /**
090         * Tracks the end of a running interaction.
091         * 
092         * @param x the current x position
093         * @param y the current y position
094         * @param e the mouse event which stoped the start interaction
095         * @param v the view in which the interaction is performed
096         */
097        void stopInteraction(int x, int y, MouseEvent e, DrawView v);
098}
