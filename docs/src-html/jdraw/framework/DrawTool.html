006package jdraw.framework;
007
008import java.awt.Cursor;
009import java.awt.event.MouseEvent;
010
011import javax.swing.Icon;
012
013/**
014 * A tool defines a mode of the drawing view, intended to be set over a tool
015 * bar. An example of a tool is the construction of new figures. Whenever a new
016 * tool is activated or deactivated, the corresponding methods are called. This
017 * can be used to e.g. adjust the mouse cursor depending on the current tool.
018 * <P>
019 * All input events targeted to the drawing view are forwarded to its current
020 * tool.
021 * 
022 * @see Figure
023 * @author Dominik Gruntz &amp; Christoph Denzler
024 * @version 2.5
025 */
026public interface DrawTool {
027
028        /**
029         * Activates the tool for the given view. This method is called whenever the
030         * user switches to this tool. Use this method to reinitialize a tool.
031         */
032        void activate();
033
034        /**
035         * Deactivates the tool. This method is called whenever the user switches to
036         * another tool. Use this method to do some clean-up when the tool is
037         * switched. Subclasses should always call {@code super.deactivate()}.
038         */
039        void deactivate();
040
041        /**
042         * Handles mouse down events in the drawing view.
043         * 
044         * @param x x coordinate of mouse position
045         * @param y y coordinate of mouse position
046         * @param e mouse event, contains state of modifiers
047         */
048        void mouseDown(int x, int y, MouseEvent e);
049
050        /**
051         * Handles mouse drag events in the drawing view.
052         * 
053         * @param x x coordinate of mouse position
054         * @param y y coordinate of mouse position
055         * @param e mouse event, contains state of modifiers
056         */
057        void mouseDrag(int x, int y, MouseEvent e);
058
059        /**
060         * Handles mouse up in the drawing view.
061         * 
062         * @param x x coordinate of mouse position
063         * @param y y coordinate of mouse position
064         * @param e mouse event, contains state of modifiers
065         */
066        void mouseUp(int x, int y, MouseEvent e);
067
068        /**
069         * Returns the cursor to be used by the draw view if this draw tool is set.
070         * 
071         * @return cursor for this draw tool.
072         */
073        Cursor getCursor();
074
075        /**
076         * Obtain an icon for this tool. This icon will be displayed in the toolbar.
077         * 
078         * @return an icon representing the tool.
079         */
080        Icon getIcon();
081
082        /**
083         * Obtain the name of this tool. This name will be used for menu entries.
084         * 
085         * @return the tool's name.
086         */
087        String getName();
088}
