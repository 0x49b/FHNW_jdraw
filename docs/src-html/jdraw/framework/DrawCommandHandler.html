006package jdraw.framework;
007
008/**
009 * The interface DrawCommandHandler defines the handler which allows to
010 * access the command history provided by a draw model. The list of
011 * commands can be iterated through using methods undo and redo.
012 *
013 * @see DrawCommandHandler
014 *
015 * @author  Dominik Gruntz &amp; Christoph Denzler
016 * @version 2.5
017 */
018
019public interface DrawCommandHandler {
020
021        /**
022         * Adds a new command to the command list at the current position. The
023         * added command is not executed by this method.
024         * Commands which were stored behind the current position are discarded.
025         * @param cmd draw command which is added to command list
026         */
027        void addCommand(DrawCommand cmd);
028
029        /**
030         * Undoes the last command in the command list. This command can be
031         * redone with operation redo unless new commands are added to the
032         * command list.
033         * @see #redo
034         */
035        void undo();
036
037        /**
038         * Redoes the next command in the command list.
039         * @see #undo
040         */
041        void redo();
042
043        /**
044         * Indicates whether an undo operation is possible.
045         * @return whether undo is possible
046         * @see #undo
047         */
048        boolean undoPossible();
049
050        /**
051         * Indicates whether a redo operation is possible.
052         * @return whether redo is possible
053         * @see #redo
054         */
055        boolean redoPossible();
056
057        /**
058         * Opens a script. Commands added with method addCommand after a script has
059         * been opened are collected until endScript is called. These command are
060         * then combined to a macro command.
061         * @see #endScript
062         */
063        void beginScript();
064
065        /**
066         * Closes a script. All commands added since the last open beginScript call
067         * are combined to a macro command.
068         * @see #beginScript
069         */
070        void endScript();
071
072        /**
073         * Clears the command list.
074         */
075        void clearHistory();
076}
