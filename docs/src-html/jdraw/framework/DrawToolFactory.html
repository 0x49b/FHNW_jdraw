006package jdraw.framework;
007
008
009/**
010 * A DrawTool factory creates an instance of a draw tool and returns an icon
011 * which can be used in the tool palette of the graphics editor.
012 *
013 * @see DrawTool
014 *
015 * @author  Dominik Gruntz &amp; Christoph Denzler
016 * @version 2.5
017 */
018public interface DrawToolFactory {
019
020        /**
021         * Returns the name of the created tool. If this value is not null, then
022         * it will overwrite the name of the created tool
023         * @see DrawTool#getName()
024         * 
025         * @return draw tool name
026         */
027        String getName();
028
029        /**
030         * Sets the name to be used for the created draw tool. This name overwrites
031         * the name of the draw tool implementation.
032         * 
033         * @param name draw tool name
034         */
035        void setName(String name);
036
037        /**
038         * Returns the path of the draw tool icon. If this value is not null, then
039         * it will overwrite the path of the created tool.
040         * @see DrawTool#getIcon()
041         * 
042         * @return draw tool name
043         */
044        String getIconName();
045
046        /**
047         * Sets the path of the draw tool icon to be used for the created draw tool.
048         * This name overwrites the name of the draw tool implementation.
049         * 
050         * @param name draw tool icon name
051         */
052        void setIconName(String name);
053
054        /**
055         * Returns a draw tool operating on the given controller.
056         * 
057         * @param context draw context
058         * @return draw tool
059         */
060        DrawTool createTool(DrawContext context);
061}
