006package jdraw.framework;
007
008import java.util.stream.Stream;
009
010/**
011 * The class DrawModel represents the model of a drawing, i.e.
012 * all figures stored in a graphic. Every draw view refers to a
013 * model.
014 *
015 * @see DrawView
016 *
017 * @author  Dominik Gruntz &amp; Christoph Denzler
018 * @version 2.5
019 */
020public interface DrawModel {
021
022        /**
023         * Adds a new figure to the draw model.
024         * 
025         * @param f figure to be added to draw model.
026         */
027        void addFigure(Figure f);
028
029        /**
030         * Removes a given figure from the draw model.
031         * 
032         * @param f figure to be removed from draw model.
033         */
034        void removeFigure(Figure f);
035        
036        /**
037         * Remove all figures from the draw model. This method is used when loading a drawing from a file
038         * in order to clear the old drawing before loading the new one.
039         */
040        void removeAllFigures();
041
042        /**
043         * Returns a sequential {@code Stream} of figures with this draw model as its
044         * source. This stream can be used to iterate over all figures contained in this
045         * model. The order of the figures is the order in which the figures were added
046         * to the model (insertion-order). The order is interpreted by the view as
047         * "back-to-front".
048         * 
049         * @return a sequential {@code Stream} over the figures in this draw model
050         */
051        Stream<Figure> getFigures();
052
053        /**
054         * Adds the specified model listener to receive model events from this draw
055         * model.
056         * 
057         * @param listener the draw model listener.
058         * @see DrawModelListener
059         */
060        void addModelChangeListener(DrawModelListener listener);
061
062        /**
063         * Removes the specified model listener so that it no longer receives model
064         * events from this draw model. This method performs no function, nor does
065         * it throw an exception, if the listener specified by the argument was not
066         * previously added to this figure.
067         * 
068         * @param listener the draw model listener.
069         * @see DrawModelListener
070         */
071        void removeModelChangeListener(DrawModelListener listener);
072
073        /**
074         * Returns the draw command handler provided by this model.
075         * 
076         * @see DrawCommandHandler
077         * @return the draw command handler used.
078         */
079        DrawCommandHandler getDrawCommandHandler();
080
081        /**
082         * Sets the index of a given figure. The order of the other figures in the
083         * model remains unchanged.
084         * 
085         * If the figure is moved to a new place, then the model sends a
086         * <code>DRAWING_CHANGED</code> event to all registered model listeners.
087         * 
088         * @param f
089         *            the figure whose index has to be set
090         * @param index
091         *            the position at which the new figure should appear. The other
092         *            figures are moved away.
093         * 
094         * @throws IllegalArgumentException
095         *             if the figure is not contained in the model.
096         * @throws IndexOutOfBoundsException
097         *             if the index is out of range (
098         *             <code>index &lt; 0 || index &ge; size()</code>) where size() is
099         *             the number of figures contained in the model.
100         */
101        void setFigureIndex(Figure f, int index) throws IllegalArgumentException, IndexOutOfBoundsException;
102
103}
