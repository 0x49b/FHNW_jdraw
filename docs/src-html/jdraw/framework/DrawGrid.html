006package jdraw.framework;
007
008import java.awt.Point;
009
010/**
011 * A DrawGrid is used to restrict the coordinates used when the mouse is
012 * clicked. The methods defined in this interface are used by the draw view.
013 * 
014 * @author Dominik Gruntz &amp; Christoph Denzler
015 * @version 2.5
016 */
017public interface DrawGrid {
018        
019        /**
020         * Returns constrained coordinates for p, e.g. rounded to a grid.
021         * 
022         * @param p mouse coordinates
023         * @return constrained coordinates
024         */
025        Point constrainPoint(Point p);
026
027        /**
028         * Returns the horizontal step size when the selection is moved with the arrow
029         * keys.
030         * 
031         * @param right true if selection is moved right; false otherwise
032         * @return step size in horizontal direction (positive result)
033         */
034        int getStepX(boolean right);
035
036        /**
037         * Returns the vertical step size when the selection is moved with the arrow
038         * keys.
039         * 
040         * @param down true if selection is moved down; false otherwise
041         * @return step size in vertical direction (positive result)
042         */
043        int getStepY(boolean down);
044        
045        /**
046         * Activates the grid. This method is called whenever method setGrid is called
047         * on a draw view.
048         */
049        void activate();
050
051        /**
052         * Deactivates the grid. This method is called whenever another grid is
053         * installed. Use this method to do some clean-up when the grid is switched.
054         */
055        void deactivate();
056
057        /**
058         * Indicates that a mouse interaction was just started. This method might be
059         * used to setup data used during this mouse interaction.
060         */
061        void mouseDown();
062
063        /**
064         * Indicates that a mouse interaction has been finished. This method might be
065         * used to clean-up data created in method mouseDown.
066         */
067        void mouseUp();
068
069}
